#include <iostream>
#include <cstdio>
#include <cstring>
#include <cstdlib>
using namespace std;

int main()
{
	//freopen("D:\\t.txt", "r", stdin);
	int m, t, u, f, d, go = 0, back = 0, ans = 0;
	scanf("%d%d%d%d%d", &m, &t, &u, &f, &d);
	for (int i = 0; i < t; i++)
	{
		char ch;
		getchar();
		scanf("%c", &ch);
		switch (ch)
		{
		case 'u':
			go += u;
			back += d;
			break;
		case 'f':
			go += f;
			back += f;
			break;
		case 'd':
			go += d;
			back += u;
			break;
		}
		if (go + back <= m)
			ans++;
		else
			break;
	}
	printf("%d\n", ans);
	return 0;
}
