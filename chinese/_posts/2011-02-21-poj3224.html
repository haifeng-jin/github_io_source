#include <cstdio>
#include <iostream>
#include <cstdlib>
#include <cstring>
using namespace std;

#define maxn 101

int n;
int form[maxn][maxn];
int out[maxn], win[maxn];
bool map[maxn][maxn];
bool visited[maxn];
int q[maxn], front = 0, rear = 0;

void init()
{
    memset(map, 0, sizeof(map));
    memset(out, 0, sizeof(out));
    memset(win, 0, sizeof(win));
    memset(visited, 0, sizeof(visited));
    scanf("%d", &n);
    int a;
    int best = 0, besti = 0;
    for (int i = 0; i < n; i++)
    {
        for (int j = 0; j < n; j++)
        {
            scanf("%d", &a);
            if (a == 3)
                win[i]++;

        }
        if (win[i] > best)
        {
            best = win[i];
            besti = i;
        }
    }
    cout << besti + 1 << endl;
}

int main()
{
    //freopen("D:\\t.txt", "r", stdin);
    init();
    return 0;
}
