简单模拟

#include <cstdio>
#include <cstring>
#include <algorithm>
using namespace std;

int n;
char st[300];

int main()
{
    while (scanf("%d", &n), n)
    {
        scanf("%s", st);
        int len = strlen(st) / n;
        for (int i = 1; i < len; i += 2)
            reverse(st + n * i, st + n * i + n);
        for (int j = 0; j < n; j++)
            for (int i = 0; i < len; i++)
                putchar(st[i * n + j]);
        putchar('\n');
    }
    return 0;
}
