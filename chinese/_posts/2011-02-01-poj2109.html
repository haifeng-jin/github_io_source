#include <iostream>
#include <cstdio>
#include <cstring>
#include <cstdlib>
#include <cmath>
using namespace std;

int main()
{
	double p, n;

	//freopen("D:\\t.txt", "r", stdin);
	while (cin >> n >> p)
		printf("%.0f\n", pow(p, 1 / n));
	return 0;
}
