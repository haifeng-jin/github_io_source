#include <iostream>
#include <cstdio>
#include <cstdlib>
#include <cstring>
using namespace std;

#define maxn 105

int n, m;
int v[maxn][maxn], f[maxn][maxn];

int main()
{
    //freopen("t.txt", "r", stdin);
    scanf("%d%d", &n, &m);
    for (int i = 1; i <= n; i++)
        for (int j = 1; j <= m; j++)
            scanf("%d", &v[i][j]);
    f[1][1] = v[1][1];
    for (int i = 2; i <= m; i++)
        f[1][i] = max(v[1][i], f[1][i - 1]);
    for (int i = 2; i <= n; i++)
    {
        f[i][i] = f[i - 1][i - 1] + v[i][i];
        for (int j = i + 1; j <= m; j++)
            f[i][j] = max(f[i - 1][j - 1] + v[i][j], f[i][j - 1]);
    }
    printf("%d\n", f[n][m]);
    return 0;
}
