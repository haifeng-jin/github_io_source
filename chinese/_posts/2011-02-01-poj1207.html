简单题，只要对所有的数字求出cycle number即可

 

#include <iostream>
#include <cstdio>
#include <cstring>
#include <cstdlib>
using namespace std;

int getcycle(int a)
{
	int r = 1;

	while (a != 1)
	{
		if (a % 2 == 1)
			a = a * 3 + 1;
		else
			a /= 2;
		r++;
	}
	return r;
}

int main()
{
	int a, b;
	//freopen("D:\\t.txt", "r", stdin);
	while (scanf("%d%d", &a, &b) != EOF)
	{
		int s = a;
		int e = b;
		if (s > e)
			swap(s, e);
		int ans = 0;
		for (int i = s; i <=e; i++)
		{
			int temp = getcycle(i);
			if (temp > ans)
				ans = temp;
		}
		printf("%d %d %d\n", a, b, ans);
	}
	return 0;
}
