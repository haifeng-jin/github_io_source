几何公式

#include <iostream>
#include <cstdlib>
#include <cstdio>
#include <cstring>
#include <cmath>
using namespace std;

#define pi acos(-1)

int D, V;

int main()
{
    while (scanf("%d%d", &D, &V), D | V)
    {
        printf("%.3f\n", pow(D * D * D - 6 * V / pi, 1.0 / 3));
    }
    return 0;
}
