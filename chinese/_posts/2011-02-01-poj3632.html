水题，求最大值最小值

 

#include <iostream>
#include <cstdio>
#include <cstring>
#include <cstdlib>
using namespace std;

int main()
{
	//freopen("D:\\t.txt", "r", stdin);
	int t;
	scanf("%d", &t);
	while (t--)
	{
		int mind, maxd, a, n;
		scanf("%d", &n);
		scanf("%d", &a);
		mind = a;
		maxd = a;
		for (int i = 1; i < n; i++)
		{
			scanf("%d", &a);
			if (mind > a)
				mind = a;
			if (maxd < a)
				maxd = a;
		}
		printf("%d\n", (maxd - mind) * 2);
	}
	return 0;
}
