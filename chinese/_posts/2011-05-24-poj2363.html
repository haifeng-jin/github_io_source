#include <iostream>
#include <cstdio>
#include <cstdlib>
#include <cstring>
using namespace std;

int cal(int a, int b, int c)
{
    return (a * b + b * c + c * a) * 2;
}

int work(int a)
{
    int ans = 1000000000;
    for (int i = 1; i * i * i <= a; i++)
        for (int j = i; i * j * j <= a; j++)
        {
            int k = a / i / j;
            if (i * j * k == a && cal(i, j, k) < ans)
                ans = cal(i, j, k);
        }
    return ans;
}

int main()
{
    freopen("t.txt", "w", stdout);
    for (int i = 1; i <= 1000; i++)
    {
        printf("%d,", work(i));
    }
    return 0;
}
