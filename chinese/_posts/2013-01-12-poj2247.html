char* make(int a, char *st)
{
    if (a % 10 == 1 && a % 100 != 11)
        strcpy(st, "st");
    else if (a % 10 == 2 && a % 100 != 12)
        strcpy(st, "nd");
    else if (a % 10 == 3 && a % 100 != 13)
        strcpy(st, "rd");
    else
        strcpy(st, "th");
    return st;
}

    f[1] = 1;

    fac[0] = 2;
    fac[1] = 3;
    fac[2] = 5;

    for (int i = 2; i <= 5842; i++)

                f[i] = f[pos[j]] * fac[j];
        for (int j = 0; j < 4; j++)
            if (f[pos[j]] * fac[j] <= f[i])
