int f[6];

{

    while (t--)
    {
        for (int i = 0; i < 4; i++)
            scanf("%d", &f[i]);
        if (f[3] - f[2] == f[2] - f[1])
            f[4] = f[3] + f[3] - f[2];
        else
            f[4] = f[3] * f[3] / f[2];
        printf("%d", f[0]);
        for (int i = 1; i < 5; i++)
            printf(" %d", f[i]);
        printf("\n");
    }
    return 0;
}
