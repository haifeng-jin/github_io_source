#include <iostream>
#include <cstdlib>
#include <cstdio>
#include <cstring>
using namespace std;

int n;

int main()
{
    int t = 0;
    while (scanf("%d", &n) && n)
    {
        t++;
        printf("%d. ", t);
        if (n & 1)
            printf("odd ");
        else
            printf("even ");
        printf("%d\n", n / 2);
    }
    return 0;
}
