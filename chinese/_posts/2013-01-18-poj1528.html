void output(int n)
{
    int i;
    int mid=n/2;
    int sum=1;
    for(i=2;i<=mid;i++)
    {
        if(n%i==0)
        {
            sum+=i;
        }
    }
    if(sum<n)
        cout<<setw(5)<<n<<"  "<<level[0]<<endl;
    else if(sum==n)
        cout<<setw(5)<<n<<"  "<<level[1]<<endl;
    else cout<<setw(5)<<n<<"  "<<level[2]<<endl;
}
