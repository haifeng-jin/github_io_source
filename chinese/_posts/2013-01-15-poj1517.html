#include <iostream>
#include <cstdio>
#include <cstdlib>
#include <cstring>
using namespace std;

double cal(int a)
{
    int temp = 1;
    double ret = 0;
    for (int i = 1; i <= a; i++)
    {
        temp *= i;
        ret += 1.0 / temp;
    }
    return ret + 1; 
}
