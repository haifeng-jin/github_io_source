#include <iostream>
#include <cstdio>
#include <cstdlib>
#include <cstring>
using namespace std;

int main()
{
    int t;
    scanf("%d", &t);
    for (int i = 0; i < t; i++)
    {
        printf("Scenario #%d:\n", i + 1);
        int a, b;
        scanf("%d%d", &a, &b);
        if ((a & 1) && (b & 1))
            printf("%d.41\n", a * b);
        else
            printf("%d.00\n", a * b);
        putchar('\n');
    }
    return 0;
}
