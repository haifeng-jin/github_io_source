题意：这题的本质是给定若干条边，问能否用其中的一些组成多边形。

分析：先排序，看前i个的和是否大于第i+1个，若大于则可以

View Code 
#include <iostream>
#include <cstdlib>
#include <cstring>
#include <cstdio>
#include <algorithm>
using namespace std;

#define maxn 30

double f[maxn];
int n;

void input()
{
    for (int i = 0; i < n; i++)
        scanf("%lf", &f[i]);
}

void work()
{
    double sum = 0;
    for (int i = 0; i < n; i++)
    {
        if (sum >= f[i])
        {
            printf("YES\n");
            return;
        }
        sum += f[i];
    }
    printf("NO\n");
}

int main()
{
    //freopen("t.txt", "r", stdin);
    while (scanf("%d", &n), n)
    {
        input();
        sort(f, f + n);
        work();
    }
    return 0;
}
