#include <iostream>
#include <cstdlib>
#include <cstdio>
#include <cstring>
using namespace std;

int main()
{
    //freopen("t.txt", "r", stdin);
    double n, m;
    scanf("%lf", &m);
    while (scanf("%lf", &n), n != 999)
    {
        printf("%.2f\n", n - m);
        m = n;
    }
    printf("End of Output\n");
    return 0;
}
