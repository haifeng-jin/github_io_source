不重复的全排列，可以使用stl中的next_permutation

View Code 
#include<iostream>
#include<algorithm>
#include<stdio.h>
char a[250];
using namespace std;
int main()
{
    int len;
    scanf("%s",a);
    len=strlen(a);
    sort(a,a+len);
    do
    {
        printf("%s\n",a);
    }
    while(next_permutation(a,a+len));
    return 0;
}
