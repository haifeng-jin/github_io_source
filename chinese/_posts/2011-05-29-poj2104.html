#include <iostream>
#include <cstdio>
#include <cstdlib>
#include <cstring>
#include <algorithm>
using namespace std;

#define maxn 100005

struct Elem
{
    int p, v;
}f[maxn];

bool operator < (const Elem &a, const Elem &b)
{
    return a.v < b.v;
}

int main()
{
//    freopen("t.txt", "r", stdin);
    int n, m;
    scanf("%d%d", &n, &m);
    for (int i = 0; i < n; i++)
    {
        scanf("%d", &f[i].v);
        f[i].p = i + 1;
    }
    sort(f, f + n);
    while (m--)
    {
        int a, b, c;
        scanf("%d%d%d", &a, &b, &c);
        int pos = -1;
        for (int i = 0; i < c; i++)
        {
            pos++;
            while (f[pos].p > b || f[pos].p < a)
                pos++;
        }
        printf("%d\n", f[pos].v);
    }
    return 0;
}
