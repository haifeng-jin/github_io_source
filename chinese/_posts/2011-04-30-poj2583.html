#include <iostream>
#include <cstdlib>
#include <cstring>
#include <cstdio>
using namespace std;

int main()
{
    //freopen("D:\\t.txt", "r", stdin);
    int c, abc, abc2, a, b;
    while (scanf("%d%d%d", &c, &abc, &abc2)!= EOF)
    {
        a = (abc2 - c - (abc - c) * 2) / 2;
        b = abc - a - c;
        printf("%d %d %d\n", a * 9 + b * 3 + c, a * 16 + b * 4 + c, a * 25 + b * 5 + c);
    }
    return 0;
}
