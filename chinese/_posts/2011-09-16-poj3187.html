#include <cstring>
#include <cstdio>

using namespace std;

#define maxn 15

int n, f[maxn], g[maxn], sum;

int cal()
{
    int ret = 0;
    for (int i = 0; i < n; i++)
        ret += f[i] * g[i];
    return ret;
}

{

    for (int i = 1; i < n; i++)
    {
        for (int j = i; j >= 1; j--)
            f[j] = f[j] + f[j - 1];
