简单题

#include <cstdio>
using namespace std;

int main()
{
    int t;
    scanf("%d", &t);
    while (t--)
    {
        long long n;
        scanf("%lld", &n);
        long long temp = 5;
        long long ans = 0;
        while (temp <= n)
        {
            ans += n / temp;
            temp *= 5;
        }
        printf("%lld\n", ans);
    }
    return 0;
}
