#include <iostream>
#include <cstdio>
#include <cstdlib>
#include <cstring>
#include <cmath>
using namespace std;

#define maxn 105
#define maxm 300

int n, x, d;
char s[maxn], p[maxn], c[maxn], m[maxn];
int spos[maxm];

void input()
{
    scanf("%s%s%s", s, p, c);
}

void make(char *st, int *pos)
{
    int len = strlen(st);
    for (int i = 0; i < len; i++)
        pos[st[i]] = i;
}

void work()
{
    n = strlen(c);
    d = (int(pow(n, 1.5) + x)) % n;
    m[d] = p[spos[c[d]]];
    for (int i = d + n - 1; i > d; i--)
    {
        int t = i % n;
        int a = spos[c[t]];
        int b = spos[m[(t + 1) % n]];
        m[t] = p[a ^ b];
    }
    m[n] = '\0';
}

int main()
{
    //freopen("t.txt", "r", stdin);
    while (scanf("%d", &x), x)
    {
        input();
        make(s, spos);
        work();
        printf("%s\n", m);
    }
    return 0;
}
