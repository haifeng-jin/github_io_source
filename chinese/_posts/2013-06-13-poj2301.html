简单题

#include <cstdio>

int main()
{
    int t;
    scanf("%d", &t);
    while (t--)
    {
        int a, b, c, d;
        scanf("%d%d", &a, &b);
        c = (a + b) / 2;
        d = (a - b) / 2;
        if (a == c + d && b == c - d && c >= 0 && d >= 0)
            printf("%d %d\n", c, d);
        else
            printf("impossible\n");
    }
    return 0;
}
