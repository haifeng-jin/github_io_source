#include <iostream>
#include <cstdlib>
#include <cstdio>
#include <cstring>
using namespace std;

int main()
{
    //freopen("t.txt", "r", stdin);
    int t;
    scanf("%d", &t);
    while (t--)
    {
        int x, y;
        scanf("%d%d", &x, &y);
        if (!((x == y) || (x - y == 2)))
        {
            puts("No Number");
            continue;
        }
        if ((x == y) ^ ((x + y) % 4 != 0))
            printf("%d\n", x + y);
        else
            printf("%d\n", x + y - 1);
    }
    return 0;
}
