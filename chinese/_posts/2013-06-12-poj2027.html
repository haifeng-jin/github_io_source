#include <cstdio>

int main()
{
    int t;
    scanf("%d", &t);
    while (t--)
    {
        int a, b;
        scanf("%d%d", &a, &b);
        if (a >= b)
            puts("MMM BRAINS");
        else
            puts("NO BRAINS");
    }
    return 0;
}
